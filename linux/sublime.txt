
tar vxjf Sublime\ Text\ 2.0.2\ x64.tar.bz2
sudo mv Sublime\ Text\ 2 /opt/sublime_text_2
sudo chown -R root:root /opt/sublime_text_2
sudo ln -s /opt/sublime_text_2/sublime_text /usr/bin/sublime

sudo bash -c "cat > /usr/share/applications/sublime.desktop" <<EOF
[Desktop Entry]
Version=2.0.2
Name=Sublime Text 2
GenericName=Text Editor
Exec=sublime
Terminal=false
Type=Application
StartupNotify=true
MimeType=text/plain;
Icon=/opt/sublime_text_2/Icon/48x48/sublime_text.png
Categories=TextEditor;IDE;Development
X-Ayatana-Desktop-Shortcuts=NewWindow

[NewWindow Shortcut Group]
Name=New Window
Exec=sublime -n
TargetEnvironment=Unity
EOF

sudo sed -i 's/gedit.desktop/sublime.desktop/g' /usr/share/applications/defaults.list
