
# install
curl -L http://install.ohmyz.sh | sh

#config
echo "Copying your current PATH and adding it to the end of ~/.zshrc for you."
sed -i -e "/export PATH=/ c\\
export PATH=\"$PATH\"
" ~/.zshrc

echo "Using the Oh My Zsh template file and adding it to ~/.zshrc"
cp ~/.oh-my-zsh/templates/zshrc.zsh-template ~/.zshrc sed -i -e "/^ZSH=/ c\\ ZSH=$ZSH
" ~/.zshrc
sed -i -e "/^plugins=(git)/ c\\
plugins=(git autojump mvn)
" ~/.zshrc
