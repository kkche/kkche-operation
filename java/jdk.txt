
sudo apt-get install openjdk-6-jdk
update-java-alternatives -l

cd /tmp
tar xvzf jdk1.7.0_51-linux-x64.gz
sudo mv jdk1.7.0_51 /opt
sudo chown -R root:root /opt/jdk1.7.0_51

sudo vi /etc/profile.d/java.sh

export PATH=$JAVA_HOME/bin:$PATH
export CLASSPATH=$CLASSPATH:$JAVA_HOME/lib/
export JAVA_OPTS="-Xmx8192m -XX:MaxPermSize=4096m"

source /etc/profile.d/java.sh

sudo update-alternatives --install /usr/bin/jps jps /opt/jdk1.7.0_51/bin/jps 1081
