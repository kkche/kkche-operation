
cd /tmp
wget http://downloads.typesafe.com/akka/akka-2.3.2.zip
unzip akka-2.3.2.zip
sudo mv akka-2.3.2 /opt/
sudo chown -R root:root /opt/akka-2.3.2/
sudo chmod -R u+x /opt/akka-2.3.2/bin/

sudo vi /etc/profile.d/scala.sh
export AKKA_HOME=/opt/akka-2.3.2
export PATH=$AKKA_HOME/bin:$PATH

source /etc/profile.d/scala.sh
